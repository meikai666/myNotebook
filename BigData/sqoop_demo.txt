----------MySQL ---> Hive---------
# 删除hive表
hive -e "drop table xxxxx"
# 创建hive表
sqoop create-hive-table --connect jdbc:mysql://xxxxxx:3306/xxx --username xx --password xxxxx --table xx --hive-table xxxx --fields-terminated-by ',' --hive-overwrite
# 导入数据
sqoop import -D mapred.job.queue.name=important --connect jdbc:mysql://xxxx:3306/xxx --username xx --password xxxxxx --table xxxxx --hive-table xxxx --hive-import --fields-terminated-by ',' --hive-overwrite -m 1
# 查看job
sqoop job --show adslot_job
# 执行job
sqoop job --exec adslot_job
